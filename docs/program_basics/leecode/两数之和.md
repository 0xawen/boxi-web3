# 两数之和

01

```rust
use std::collections::HashMap;

impl Solution {
    pub fn two_sum(nums: Vec<i32>, target: i32) -> Vec<i32> {
        // hashmap
        let mut map = HashMap::new();
        for (index, value) in nums.iter().enumrate() {
            if map.contains_key(value) {
                // 判断是否存在目标值
                return vec![index as i32, map[value]];
            }
            
            // x 是需要找到的另一个值
            let x = target - nums[index];
            map.insert(x, index as i32);
        }
        // 增加一个不可达宏，
        unreachable!()
    }
}
```
